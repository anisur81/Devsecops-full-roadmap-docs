```yaml
name: Secret Scan
on:
  workflow_call:
jobs:
  gitleaks:
    runs-on: ubuntu-latest
    steps:

     # Gitleaks Scanning
     
      - uses: actions/checkout@v4
      - name: Run Gitleaks
        uses: gitleaks/gitleaks-action@v1
        env:
           GITLEAKS_EXIT_CODE: 1

# Add SARIF upload to GitHub Security tab
      - name: Upload Gitleaks SARIF
        uses: github/codeql-action/init@v4
        with:
           languages: python
      - name: sarif file auto upload    
        uses: github/codeql-action/analyze@v4
    
        # TRUFFLEHOG Scanning
        
      - name: Run TruffleHog Scan
        uses: trufflesecurity/trufflehog@v3.82.8
        with:
           path: ./
           base: ${{ github.event.before }}
           head: ${{ github.sha }}  
```

# 04-SCA-dependency-scan.yml

```yaml
name: SCA-Source Code Composition Analysis

on:
  workflow_call:

jobs:
  dependency-scan:
    name: SCA with trivy-semgrep-pipaudit
    runs-on: ubuntu-latest

   # Job 4: software composition analysis (SCA) (trivy + semgrep + pipaudit) 
    steps:
     - name: Checkout code
       uses: actions/checkout@v4

    # Trivy for SCA
     - name: Install Trivy
       run: |
          sudo apt-get update
          sudo apt-get install -y wget
          TRIVY_VERSION="0.71.0"
          wget https://github.com/aquasecurity/trivy/releases/download/v${TRIVY_VERSION}/trivy_${TRIVY_VERSION}_Linux-64bit.tar.gz
          tar -xzf trivy_${TRIVY_VERSION}_Linux-64bit.tar.gz
          sudo mv trivy /usr/local/bin/
          
     - name: Verify installation
       run: trivy --version
      
     - name: Run Trivy Filesystem Scan
       run: |
          trivy fs . --severity HIGH,CRITICAL --exit-code 1

    # Semgrep for SCA 
     - name: Run Semgrep
       uses: returntocorp/semgrep-action@v1
       with:
          config: p/owasp-top-ten

     - name: Upload Semgrep report
       uses: actions/upload-artifact@v4
       with:
          name: semgrep-report
          path: semgrep-results.json
   
   # production-ready SCA job using pip-audit
     - name: Install dependencies
       run: |
          pip install --upgrade pip
          pip install pip-audit

     - name: Run SCA Scan (pip-audit)
       run: |
          pip-audit -r requirements.txt -s osv || true
          if [ $? -ne 0 ]; then
           echo "High severity vulnerabilities found"
           exit 1
          fi
```

  PYTHON_VERSION: '3.12.3'

          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Install Snyk CLI
        run: npm install -g snyk

        run: snyk auth ${{ secrets.SNYK_TOKEN }}

      - name: Test Python dependencies
        run: snyk test --file=requirements.txt

      - name: Monitor project (optional)
        run: snyk monitor --file=requirements.txt

# 05-SAST-security.yml

```yaml
name: SAST secuity Scan
on:
  workflow_call:
   secrets:
      SONAR_TOKEN:
        required: true

env:
  PYTHON_VERSION: '3.12.3'
  
jobs:

# Job 3: Static application security testing (SAST) 
  security-scan:
    name: Security Scanning SAST
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: ${{ env.PYTHON_VERSION }}
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
        pip install safety bandit django pytest-django coverage
        pip install setuptools wheel checkov cryptography pyOpenSSL boto3 botocore urllib3

    # Safety Python dependency vulnerability scanner
    - name: Check for security vulnerabilities
      run: PYTHONWARNINGS="ignore" safety check -r requirements.txt --json > safety-report.json

    - name: Upload Safety Reports
      uses: actions/upload-artifact@v4
      with:
         name: safety-reports
         path: |
             safety-report.json 
    
    # Bandit with OWASP check specific for python
    - name: Run Bandit security scan
      run: bandit -r . -ll -f json -o bandit-report.json
    - name: Upload bandit Reports
      uses: actions/upload-artifact@v4
      with:
         name: bandit-reports
         path: | 
            bandit-report.json
            
    # Semgrep Enterprise SAST 
    - name: Run Semgrep
      uses: returntocorp/semgrep-action@v1
      with:
        config: "p/security-audit"
   # Codeql  initialize
    - name: Initialize CodeQL
      uses: github/codeql-action/init@v3
      with:
          languages: python
          queries: security-extended,security-and-quality   
     # Codeql code analysis     
    - name: Perform CodeQL Analysis
      uses: github/codeql-action/analyze@v3

    - name: Run Checkov scan
      run: |
          checkov -d . --framework terraform,kubernetes,cloudformation \
          --output cli --soft-fail

    # Sonarqube SAST checking
    
    - name: SonarCloud Scan
      uses: SonarSource/sonarcloud-github-action@v2
      env:
          SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
      with:
          args: >
            -Dsonar.organization=anisur81
            -Dsonar.projectKey=anisur81_portfolio
            -Dsonar.projectName=django-app
            -Dsonar.sources=.
            -Dsonar.python.version=3.11
            -Dsonar.exclusions=**/migrations/**,**/venv/**,**/__pycache__/**

    - name: Run Tests with Coverage for Sonarqube
      run: |
           coverage run manage.py test --verbosity=2
           coverage xml
    - name: Upload coverage Reports
      uses: actions/upload-artifact@v4
      with:
         name: coverage-reports
         path: | 
            coverage.xml

# 07-build-unit.yml

```yaml
name: CI Pipeline (Build + Tests)
on:
  workflow_call:
env:
  PYTHON_VERSION: "3.12"
jobs:
  build:
    name: Build Application
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install Dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Build Project (Django check)
        run: |
          python manage.py check
          python manage.py migrate --noinput

  unit_tests:
    name: Unit Tests
    runs-on: ubuntu-latest
    needs: build

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}

      - name: Install Dependencies
        run: |
          pip install -r requirements.txt
          pip install pytest django pytest-django pytest-cov

      - name: Run Unit Tests
        continue-on-error: true
        run: |
          pytest --cov=. --cov-report=xml --cov-report=html
          pytest -m "unit" --disable-warnings --maxfail=1

      - name: Upload coverage to Codecov
        uses: codecov/codecov-action@v4
        with:
           files: ./coverage.xml
           flags: unittests
           name: codecov-umbrella
           fail_ci_if_error: false
    
      - name: Upload coverage reports
        uses: actions/upload-artifact@v4
        with:
           name: coverage-report
           path: htmlcov/  
```

# 09-docker-build-SBOM.yml

```yaml
name: Docker build and SBOM

permissions:
      contents: read
      packages: write
      security-events: write
      
on:
  workflow_call:
  
jobs:
# Job 6: Build Docker Image and Push to GitHub Container Registry
  docker:
    name: Build & Push Docker Image
    runs-on: ubuntu-latest
    
    # Push only on main branch
    if: github.ref == 'refs/heads/main'

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Log in to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build and Push Docker Image
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            ghcr.io/${{ github.repository }}:latest
            ghcr.io/${{ github.repository }}:${{ github.sha }}   
      
      - name: Generate SBOM
        uses: anchore/sbom-action@v0     
```

# 12-DAST-and-Smoke.yml

```yaml
name: DAST Security Checking
on:
  workflow_call:

jobs:

 k8s-test:
    name: DAST with OWASP ZAP and smoke test 
    runs-on: ubuntu-latest
  
    steps:
      - uses: actions/checkout@v4

      - name: Install Kind
        run: |
          curl -Lo ./kind https://kind.sigs.k8s.io/dl/latest/kind-linux-amd64
          chmod +x ./kind
          sudo mv ./kind /usr/local/bin/kind

      - name: Install kubectl
        run: |
          curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
          chmod +x kubectl
          sudo mv kubectl /usr/local/bin/

      - name: Create cluster
        run: |
          kind create cluster

       # Build Docker image
      - name: Build image
        run: docker build -t portfolio:${{ github.sha }} .

    # Load image into Kind
      - name: Load image into Kind
        run: kind load docker-image portfolio:${{ github.sha }} 

      - name: Check Docker image size
        run: |
           SIZE=$(docker image inspect portfolio:${{ github.sha }} --format='{{.Size}}')
           echo "Raw size: $SIZE bytes" 
     
      - name: Deploy app
        run: |
          kubectl create deployment demo \
           --image=ghcr.io/${{ github.repository }}:${{ github.sha }}
            #--image=portfolio:${{ github.sha }}
          kubectl expose deployment demo \
           --type=ClusterIP \
           --port=8000 \
           --target-port=8000

      - name: Check deployment
        run: |
          kubectl wait --for=condition=Ready nodes --all --timeout=120s
          kubectl rollout status deployment/demo --timeout=1200s
          kubectl get all
          kubectl get events --sort-by=.metadata.creationTimestamp
          kubectl get deployments
          kubectl get replicasets
          kubectl get pods -A
          kubectl get svc
          kubectl describe deployment demo
    
      - name: Port forward 
        run: |
          kubectl port-forward svc/demo 8000:8000 &
          sleep 15
      - name: Verify app
        run: |
          curl http://localhost:8000 
      - name: Run OWASP ZAP
        run: |
          docker run --network host \
          ghcr.io/zaproxy/zaproxy:stable \
          zap-baseline.py \
          -t http://localhost:8000 -I
